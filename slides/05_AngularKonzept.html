Ist das cool? 
<input id="zeigeText" type="checkbox">

$(function() {
    function showOrHide() {
        var isChecked = $('#zeigeText').is(':checked');
        
        // imperative Code dependency
        var fn = $('#einText');
        if (isChecked)
            fn.show();
        else
            fn.hide();
    }
    $('#zeigeText').change(function() {
        showOrHide();
    });
    showOrHide();
});
